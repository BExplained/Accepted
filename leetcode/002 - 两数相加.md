## 代码

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        int more = 0; // 进一
        int sum = l1->val + l2->val;
        if (sum >= 10) {
            more = 1;
            sum -= 10;
        }

        ListNode* head = new ListNode(sum);
        ListNode* cur = head;

        l1 = l1->next;
        l2 = l2->next;

        while (l1 && l2) {
            sum = l1->val + l2->val + more;
            if (sum >= 10) {
                more = 1;
                sum -= 10;
            } else {
                more = 0;
            }

            ListNode* t = new ListNode(sum);
            cur->next = t;
            cur = t;

            l1 = l1->next;
            l2 = l2->next;
        }

        while (l1) {
            sum = l1->val + more;
            if (sum >= 10) {
                more = 1;
                sum -= 10;
            } else {
                more = 0;
            }

            ListNode* t = new ListNode(sum);
            cur->next = t;
            cur = t;

            l1 = l1->next;
        }

        while (l2) {
            sum = l2->val + more;
            if (sum >= 10) {
                more = 1;
                sum -= 10;
            } else {
                more = 0;
            }

            ListNode* t = new ListNode(sum);
            cur->next = t;
            cur = t;

            l2 = l2->next;
        }

        if (more) {
            ListNode* t = new ListNode(more);
            cur->next = t;
        }

        return head;
    }
};
```
