## 第一次

```c++
class Solution {
public:
    bool isMatch(string s, string p) {
        int dp[35][35]; // dp[i][j] 表示 s[0..i) 与 p[0...j) 是否匹配
        memset(dp, 0, sizeof(dp));

        int m = s.size();
        int n = p.size();
        dp[0][0] = 1;

        for (int j = 1; j <= n; ++j) {
            if (j - 2 >= 0 && p[j - 1] == '*')
                dp[0][j] |= dp[0][j - 2];
        }

        for (int i = 1; i <= m; ++i) {
            for (int j = 1; j <= n; ++j) {
                if (p[j - 1] == '*') {
                    dp[i][j] |= dp[i][j - 1];
                    if (j - 2 >= 0 && (s[i - 1] == p[j - 2] || p[j - 2] == '.')) {
                        dp[i][j] |= dp[i - 1][j];
                    }
                }
                else if (p[j - 1] == '.') {
                    dp[i][j] |= dp[i - 1][j - 1];
                }
                else {
                    if (s[i -1] == p[j - 1]) {
                        dp[i][j] |= dp[i - 1][j - 1];
                    }
                }
            }
        }

        return dp[m][n];
    }
};
```
