## 代码

```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int> ret;
        map<int, int> hash;

        int size = nums.size();
        hash[nums[0]] = 0;
        for (int i = 1; i < size; ++i) {
            auto it = hash.find(target - nums[i]);
            if (it != hash.end()) {
                ret.push_back(it->second);
                ret.push_back(i);
                return ret;
            } else {
                hash[nums[i]] = i;
            }
        }

        return ret;
    }
};
```
