**题意：**

判断是否是一颗树。

**分析：**

只有一个树根；且不能有环。

```c++
#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <queue>
#include <stack>
#include <vector>
#include <bitset>
#include <utility>
#include <map>
#include <set>

#define MOD 100007
#define INF 1000000007

using namespace std;

#include<iostream>    
#include<string.h>  
#include<algorithm>  

using namespace std;

int pre[1005];
int in[1005];
int vis[1005];
int x, y;
int rootNum;
bool flag;

int find(int x)
{
	return (x != pre[x]) ? find(pre[x]) : x;
}

void join(int x, int y)
{
	int x_root = find(x);
	int y_root = find(y);

	if (x_root != y_root)
		pre[y_root] = x_root;
	else
		flag = false;
}

int main()
{
	int cas = 1;

	while (scanf("%d%d", &x, &y))
	{
		if (x < 0 && y < 0)
			break;
		if (x == 0 && y == 0)
		{
			printf("Case %d is a tree.\n", cas++);
			continue;
		}

		// 初始化
		memset(vis, 0, sizeof(vis));
		memset(in, 0, sizeof(in));
		for (int i = 1; i < 1005; i++) pre[i] = i;
		flag = true;
		rootNum = 0;

		// 处理输入的边
		vis[x] = vis[y] = 1;
		join(x, y);
		in[y]++;
		if (in[y] >= 2)
			flag = false;

		while (scanf("%d%d", &x, &y) && (x || y))
		{
			vis[x] = vis[y] = 1;
			join(x, y);
			in[y]++;
			if (in[y] >= 2)
				flag = false;
		}

		if (flag)
		{
			for (int i = 1; i < 1005; i++)
				if (vis[i] && i == pre[i])
					rootNum++;
		}

		if (flag && rootNum == 1)
			printf("Case %d is a tree.\n", cas++);
		else
			printf("Case %d is not a tree.\n", cas++);
	}

	return 0;
}
```
