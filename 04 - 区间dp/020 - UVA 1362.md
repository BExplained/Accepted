**题意：**

给定一序列，该序列是遍历多叉树得到的，且遍历的规则是，对于当前结点总是先遍历它的最左子树，然后依次，根据该序列求出共有多少种多叉树。

若是看不清题意，看下给出的 5 幅图，它们都是第一组样例的答案，即 ABABABA。

**分析：**

dp[i][j] 表示区间 [i, j] 可以得到的多叉树种类。

```c++
#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <queue>
#include <stack>
#include <vector>
typedef long long int ll;
const ll MOD = 1e9;
const int INF = 99999999;
using namespace std;

int n;
char s[305];
ll dp[305][305];

ll dfs(int i, int j)
{
	if (dp[i][j] != -1)
		return dp[i][j];

	if (i == j)
	{
		dp[i][j] = 1;
		return dp[i][j];
	}

	if (s[i] != s[j])
	{
		dp[i][j] = 0;
		return dp[i][j];
	}

	for (int k = i + 2; k <= j - 2; k++)
	{
		if (s[i] == s[k])
			dp[i][j] = (dp[i][j] + dp[i][k] * dp[k][j] % MOD) % MOD;
	}
}

int main()
{
	while (~scanf("%s", s + 1))
	{
		n = strlen(s + 1);

		memset(dp, -1, sizeof(dp));

		printf("%lld\n", dfs(1, n));
	}
	return 0;
}
```
