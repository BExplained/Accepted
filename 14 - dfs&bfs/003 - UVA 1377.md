**题意：**

n 个测量值，要你做一把尺子，可以测量者 n 个数，且上面的值最少最小。

**分析：**

n 最大是 50，但注意到 M 最大是 7，所以真正需测量只有 C(7, 2) 即 21 个。

对于待测量的一个数，想要利用尺子上的值去测量它（假设目前没法这个值），我们只需再新增一个刻度，这个刻度要么通过加法得到，要么通过减法得到。

```c++
#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <queue>
#include <stack>
#include <vector>
#include <bitset>
#include <utility>
#include <map>
#include <set>
using namespace std;
typedef long long int ll;
typedef pair<int, int> PII;
const int MOD = (int)1e9 + 7;
const int INF = 1000000007;
const ll INFLL = 0x3f3f3f3f3f3f3f3f;

int n;
int nn;
int id[100005];
int d[25];
int vis[1 << 22];
int maxx;
set<int> res;

struct P
{
	int st;
	set<int> ans;
};

void bfs()
{
	queue<P> Q;
	P p;
	p.st = 0, p.ans.clear(), p.ans.insert(0);
	Q.push(p);
	while (!Q.empty())
	{
		p = Q.front();
		Q.pop();
		if (p.st == ((1 << nn) - 1)) // 看看是不是所有的长度值都可以测量了，如果可以就更新最后的答案保存在 res
		{
			if (res.size() == 0)
				res = p.ans;
			else
			{
				if (res.size() < p.ans.size())
					return;
				else if (res.size() > p.ans.size())
					res = p.ans;
				else
				{
					if (*(res.rbegin()) > *(p.ans.rbegin()))
						res = p.ans;
				}
			}
		}

		if (p.ans.size() == 7) // 没必要做了
			continue;

		for (int i = 0; i < nn; i++)
		{
			if ((p.st & (1 << i)))
				continue;

			for (set<int>::iterator it = p.ans.begin(); it != p.ans.end(); it++)
			{
				if (*it > d[i])
				{
					P q = p;
					int sum = *it - d[i];

					// 新加入标尺值 sum，那么就会多出现几个可测量的值，看看是否出现在给定的值中
					for (set<int>::iterator it1 = q.ans.begin(); it1 != q.ans.end(); it1++)
					{
						int nu = abs(*it1 - sum);
						if (id[nu] == -1)
							continue;
						q.st |= (1 << id[nu]);
					}

					q.ans.insert(sum);
					if (!vis[q.st])
					{
						Q.push(q);
						vis[q.st] = 1;
					}
				}
				if (*it + d[i] <= maxx)
				{
					P q = p;
					int sum = *it + d[i];

					for (set<int>::iterator it1 = q.ans.begin(); it1 != q.ans.end(); it1++)
					{
						int nu = abs(*it1 - sum);
						if (id[nu] == -1)
							continue;
						q.st |= (1 << id[nu]);
					}

					q.ans.insert(sum);
					if (!vis[q.st])
					{
						Q.push(q);
						vis[q.st] = 1;
					}
				}
			}
		}
	}
}

int main()
{
	int c = 1;

	while (scanf("%d", &n) && n)
	{
		res.clear();
		memset(id, 0, sizeof(id));
		memset(vis, 0, sizeof(vis));

		int t;
		nn = 0;
		for (int i = 0; i < n; i++)
		{
			scanf("%d",&t);
			if (!id[t])
			{
				d[nn++] = t;
				id[t] = 1;
			}
		}

		sort(d, d + nn);
		maxx = d[nn - 1];
		memset(id, -1, sizeof(id));
		for (int i = 0; i < nn; i++)
			id[d[i]] = i;

		bfs();

		printf("Case %d:\n%d\n", c++, res.size());
		int f = 0;
		for (set<int>::iterator it = res.begin(); it != res.end(); it++)
		{
			if (f++)
				printf(" ");
			printf("%d", *it);
		}
		printf("\n");
	}
	return 0;
}
```
